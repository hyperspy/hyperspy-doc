    offsets = np.stack([np.arange(0, 100, 10)]*2).T + np.array([5,]*2)
    texts = np.array(['a', 'b', 'c', 'd', 'e', 'f', 'g', 'f', 'h', 'i'])

    m = hs.plot.markers.Texts(

        sizes=3,
        facecolor="black",
        )
    s.plot()
