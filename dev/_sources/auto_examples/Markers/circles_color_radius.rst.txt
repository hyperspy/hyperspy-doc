    sizes = rng.random((10, )) * 20 + 5

    # Define the position of the circles
    offsets = rng.random((10, 2)) * 100

    m = hs.plot.markers.Circles(

        offsets=offsets,
        linewidth=2,
        )

    s.plot()
    s.add_marker(m)
